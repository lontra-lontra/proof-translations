\def\forcingDefinition{
 26
\begin{definition}
 27
\textbf{Forcing}. When a sentence \(\phi\) of a language \(\mathcal{L}\) is \textbf{forced} by a structure \(C(p)\) of a \(\mathcal{L}\)-frame \(\mathcal{C}\), we denote it as:
 28
\(p \Vdash_{\mathcal{C}} \phi\) \\
 29
Forcing is defined by induction: \cite{book1}
 30
 31
\begin{itemize}
 32
\item \(p \Vdash_{\mathcal{C}} \phi \text{ iff } \phi \text{ is true in } C(p)\) (if \(\phi\) is an atomic sentence).
 33
\item \(p \Vdash_{\mathcal{C}} (\phi \to \psi) \text{ iff }\) for all \(q \geq p\), if \(q \Vdash_{\mathcal{C}} \phi\), then \(q \Vdash_{\mathcal{C}} \psi\).
 34
\item \(p \Vdash_{\mathcal{C}} \neg \phi \text{ iff } \) for all \(q \geq p\), \(q\) does not force \(\phi\).
 35
\item \(p \Vdash_{\mathcal{C}} (\forall x) \phi(x) \text{ iff }\) for all \(q \geq p\) and \(d\) in \(\mathcal{L}_{C(q)}\), \(q \Vdash_{\mathcal{C}} \phi(d)\).
 36
\item \(p \Vdash_{\mathcal{C}} (\exists x) \phi(x) \text{ iff }\) there exists a \(d\) in \(\mathcal{L}_{C(q)}\), such that \(p \Vdash_{\mathcal{C}} \phi(d)\).
 37
\item \(p \Vdash_{\mathcal{C}} (\phi \land \psi) \text{ iff } p \Vdash_{\mathcal{C}} \phi \text{ and } p \Vdash_{\mathcal{C}} \psi\).
 38
\item \(p \Vdash_{\mathcal{C}} (\phi \lor \psi) \text{ iff } p \Vdash_{\mathcal{C}} \phi \text{ or } p \Vdash_{\mathcal{C}} \psi\).
